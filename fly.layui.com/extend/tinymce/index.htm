使用文档见码云




    <textarea name="" id="edit" cols="30" rows="10"></textarea>
    
<script>
    layui.extend({
        tinymce: '{/}./layui/mods/tinymce/tinymce'
    }).use(['tinymce', 'util', 'layer'], function () {
        var tinymce = layui.tinymce
        var edit = tinymce.render({
            elem: "#edit"
            , height: 600
            , width:'100%'
        });
    });
</script>
